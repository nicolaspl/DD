-------- 0. Aktualizacja listy dostêpnego oprogramowania -------- 

sudo apt-get update

-------- 1. Instalacja i konfiguracja awscli -------- 

sudo apt install awscli
aws configure


-------- 2. Instalacja pythona 3.6 -------- 

sudo add-apt-repository ppa:jonathonf/python-3.6;
sudo apt-get update
sudo apt-get install python3.6;

-------- 3. Instalacja pip3 -------- 

sudo apt-get install python3-pip

-------- 4. Instalacja virtualenv -------- 

sudo python3.6 -m pip install virtualenv

-------- 5. Tworzenie projektu w virtualenv -------- 

## folder dla projektów
mkdir /home/ubuntu/virtualenv

## Tworzenie œrodowiska projektu, gdzie FBGetHSVDataEnv to przyk³adowa nazwa
virtualenv -p /usr/bin/python3.6 /home/ubuntu/virtualenv/FBGetHSVDataEnv

## Aktywacja œrodowiska
source /home/ubuntu/virtualenv/FBGetHSVDataEnv/bin/activate

## Instalacja bibliotek (kilka przyk³adowych dla funkcji FBGetHSVDataEnv)
python3.6 -m pip install boto3;
python3.6 -m pip install pandas;
python3.6 -m pip install pillow;

## Deaktywacja œrodowiska
deactivate

-------- 6. Tworzenie paczki -------- 

## Kopiowanie plików bibliotek do folderu z kodem
cp -r /home/ubuntu/virtualenv/FBGetHSVDataEnv/lib/python3.6/site-packages/* /home/ubuntu/scripts/lambda/FBGetHSVData/

## Instalacja programu zip
sudo apt install zip

## Kompresja paczki do formatu .zip
cd /home/ubuntu/scripts/lambda/FBGetHSVData/;
zip -r /home/ubuntu/scripts/lambda/FBGetHSVData.zip ./*;
cd -;


-------- 7. Wysy³anie paczki na s3 -------- 
aws s3 cp /home/ubuntu/scripts/lambda/FBGetHSVData.zip s3://deepdoccodes/produkcja/ETL\ Facebook/
